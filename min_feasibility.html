# FGSM attack implementation for a batch of inputs

def fgsm_attack(model, x, y, epsilon): # model is the trained model, x is the input tensor, y is the true label tensor, epsilon is the attack strength
    """
    FGSM attack for a batch of inputs.
    """
    x_adv = x.clone().detach().requires_grad_(True) # create a copy of the input tensor that requires gradient for attack generation

    logits = model(x_adv) # get the model's output logits for the adversarial input
    loss = nn.CrossEntropyLoss()(logits, y) # calculate the loss between the model's output and the true labels for the adversarial input

    model.zero_grad() # zero out any existing gradients in the model to ensure we only get gradients for the current input
    loss.backward() # backpropagate the loss to compute gradients with respect to the input tensor

    grad_sign = x_adv.grad.sign() # get the sign of the gradients to determine the direction of the attack for each feature
    x_adv = x_adv + epsilon * grad_sign # create the adversarial example by adding a small perturbation in the direction of the gradient sign, scaled by epsilon

    return x_adv.detach()

# Function to compute IG attributions in batches to avoid memory issues

def compute_ig_attributions(ig, inputs, target_labels, baseline, batch_size=64, n_steps=50):
    attrs = []
    for i in range(0, inputs.shape[0], batch_size):
        xb = inputs[i:i+batch_size]
        yb = target_labels[i:i+batch_size]
        bb = baseline[i:i+batch_size]

        attr = ig.attribute(
            xb,
            baselines=bb,
            target=yb,
            n_steps=n_steps
        )
        attrs.append(attr.detach())
    return torch.cat(attrs, dim=0)

ATTR_STEPS = 50
BATCH_SIZE = 64

# Experiment 2: FGSM epsilon sweep (Notebook Cell)
# Assumes you already have:
#   model (trained, eval mode OK)
#   X_eval_clean (torch.Tensor, shape [500, 78])
#   y_eval_preserved (torch.Tensor, shape [500])
#   ig (IntegratedGradients(model))
#   baseline = torch.zeros_like(X_eval_clean)
#   compute_ig_attributions(...)  # your batch-friendly IG helper from Step 19
#   fgsm_attack(...)              # your FGSM helper from Step 18
# =========================

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

from sklearn.metrics import roc_curve, roc_auc_score
from sklearn.metrics.pairwise import cosine_distances
from numpy.linalg import norm

# --- 1) Configure epsilon sweep ---
eps_list = [0.001, 0.005, 0.01, 0.02]

# IG parameters (keep fixed across eps)
ATTR_STEPS = 50
BATCH_SIZE = 64

# --- 2) Pre-compute CLEAN attributions once (important!) ---
# This keeps the experiment focused on the effect of epsilon.
attr_clean = compute_ig_attributions(
    ig,
    X_eval_clean,
    y_eval_preserved,
    baseline,
    batch_size=BATCH_SIZE,
    n_steps=ATTR_STEPS
)
A_clean = attr_clean.cpu().numpy()

# clean-vs-clean control distribution (cosine) to use in ROC
clean_self = np.array([
    cosine_distances(A_clean[i:i+1], A_clean[i:i+1])[0, 0]
    for i in range(A_clean.shape[0])
])

results = []
all_cos = {}   # store drift arrays per epsilon (optional)
all_euc = {}   # store drift arrays per epsilon (optional)

# --- 3) Sweep epsilons ---
for eps in eps_list:
    # 3.1 Generate adversarial samples
    X_adv = fgsm_attack(model, X_eval_clean, y_eval_preserved, eps)

    # 3.2 Check prediction preservation
    model.eval()
    with torch.no_grad():
        logits_adv = model(X_adv)
        preds_adv = torch.argmax(logits_adv, dim=1)

    preserved_mask = preds_adv == y_eval_preserved
    preserved_count = int(preserved_mask.sum().item())
    preserved_ratio = preserved_count / len(y_eval_preserved)

    # Filter (should usually be all, but keep correct behaviour)
    X_clean_f = X_eval_clean[preserved_mask]
    X_adv_f = X_adv[preserved_mask]
    y_f = y_eval_preserved[preserved_mask]

    # 3.3 Compute adversarial attributions for preserved subset only
    baseline_f = torch.zeros_like(X_clean_f)
    attr_adv = compute_ig_attributions(
        ig,
        X_adv_f,
        y_f,
        baseline_f,
        batch_size=BATCH_SIZE,
        n_steps=ATTR_STEPS
    )
    A_adv = attr_adv.cpu().numpy()

    # 3.4 Compute drift (cosine + euclidean)
    cos_drift = np.array([
        cosine_distances(A_clean[i:i+1], A_adv[i:i+1])[0, 0]
        for i in range(A_adv.shape[0])
    ])

    euc_drift = norm(A_clean[:A_adv.shape[0]] - A_adv, axis=1)

    all_cos[eps] = cos_drift
    all_euc[eps] = euc_drift

    # 3.5 ROC-AUC (adversarial drift vs clean-self control)
    # Labels: 1 = adversarial, 0 = clean
    labels = np.concatenate([np.ones_like(cos_drift), np.zeros_like(clean_self[:len(cos_drift)])])

    scores_cos = np.concatenate([cos_drift, clean_self[:len(cos_drift)]])
    auc_cos = roc_auc_score(labels, scores_cos)

    scores_euc = np.concatenate([euc_drift, np.zeros_like(euc_drift)])  # euclid clean-self baseline is 0
    auc_euc = roc_auc_score(labels, scores_euc)

    # 3.6 Summaries
    results.append({
        "epsilon": eps,
        "preserved_count": preserved_count,
        "preserved_ratio": preserved_ratio,
        "mean_cosine_drift": float(np.mean(cos_drift)) if len(cos_drift) else np.nan,
        "median_cosine_drift": float(np.median(cos_drift)) if len(cos_drift) else np.nan,
        "mean_euclid_drift": float(np.mean(euc_drift)) if len(euc_drift) else np.nan,
        "median_euclid_drift": float(np.median(euc_drift)) if len(euc_drift) else np.nan,
        "auc_cosine": float(auc_cos),
        "auc_euclid": float(auc_euc),
    })

df_sweep = pd.DataFrame(results).sort_values("epsilon")
display(df_sweep)
